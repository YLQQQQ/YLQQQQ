#include <bits/stdc++.h>
#define rep(i, n, m) for(int i = n; i <= m; ++i)
using namespace std;
const int N = 8e5 + 10;
int def[N], low[N], st[N], ok[N], in = 0;
int h[N], ne[N], e[N], cnt, ans = 0;
void add(int a, int b){
     e[cnt] = b, ne[cnt] = h[a], h[a] = cnt++;
}
void tarjan(int u, int fa){
    st[u] = 1;
    def[u] = low[u] = ++ in;
    int num = 0;
    for(int i = h[u]; ~i; i = ne[i]){
        int v = e[i];
        if(!st[v]){
            num ++;
            tarjan(v, u);
            low[u] = min(low[u], low[v]);
            if(fa != u && low[v] >= def[u] && !ok[u]){
                ok[u] = 1;
                ans ++;
            }
        }else if(v != fa){
            low[u] = min(low[u], def[v]);
        }
    }
    if(u == fa && num >= 2 && !ok[u]){
        ok[u] = 1;
        ans ++;
    }
}
int main()
{
    //ios::sync_with_stdio(0); cin.tie(0);
    memset(h, -1, sizeof h);
    memset(ok, 0, sizeof ok);
    int n, m; cin >> n >> m;
    rep(i, 1, m){
        int u, v; cin >> u >> v;
        add(u, v), add(v, u);
    }
    rep(i, 1, n){
        if(!st[i]){
            in = 0;
            tarjan(i, i);
        }
    }
    cout << ans << '\n';
    rep(i, 1, n){
        if(ok[i]) cout << i << ' ';
    }
    return 0;
}
